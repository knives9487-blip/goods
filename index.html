function renderData() { if (!cachedData) return; const month = document.getElementById('monthPicker').value; const list = document.getElementById('historyList'); list.innerHTML = ''; let grandTotal = 0, filteredTotal = 0, groups = {}; Object.keys(cachedData).forEach(k => { const item = cachedData[k]; if (item.date?.startsWith(month)) { grandTotal += item.amount; // --- 強化過濾邏輯：相容新舊資料結構 --- let isMatch = false; if (currentFilter === 'all') { isMatch = true; } else { // 優先比對索引 (新資料)，若無索引則比對名字 (舊資料) if (item.userIndex !== undefined) { isMatch = (item.userIndex === currentFilter); } else if (item.user !== undefined) { const targetName = (currentFilter === 1) ? uName1 : uName2; isMatch = (item.user === targetName); } } if (isMatch) { filteredTotal += item.amount; if (!groups[item.date]) groups[item.date] = []; groups[item.date].push({ ...item, id: k }); } } }); // 預算條（依據總支出計算） const percent = Math.min((grandTotal / monthlyBudget) * 100, 100).toFixed(1); document.getElementById('progress-bar').style.width = percent + '%'; document.getElementById('percent-label').innerText = `${percent}%`; document.getElementById('budget-summary').innerText = `剩餘：$${Math.max(monthlyBudget-grandTotal, 0).toLocaleString()}`; // 顯示標題總計 const filterName = currentFilter === 'all' ? '月' : (currentFilter === 1 ? uName1 : uName2); document.getElementById('totalDisplay').innerText = `${filterName}支出：$${filteredTotal.toLocaleString()}`; // 渲染明細 Object.keys(groups).sort((a, b) => new Date(b) - new Date(a)).forEach(date => { let dayHtml = '', dayTotal = 0; groups[date].sort((a,b)=>b.ts - a.ts).forEach(item => { dayTotal += item.amount; // 決定標籤顯示名稱 let displayName = ""; if (item.userIndex !== undefined) { displayName = (item.userIndex === 2) ? uName2 : uName1; } else { displayName = item.user || "匿名"; // 舊資料直接顯示存儲的名字 } dayHtml += `
${item.emoji} ${item.category} $${item.amount} ${displayName}
${item.note ? `
💬 ${item.note}
` : ''}
`; }); list.innerHTML += `
📅 ${date.slice(5)}$${dayTotal}
${dayHtml}`; }); }
