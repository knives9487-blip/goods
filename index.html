function renderData() { if (!cachedData) return; const month = document.getElementById('monthPicker').value; const list = document.getElementById('historyList'); list.innerHTML = ''; let grandTotal = 0, filteredTotal = 0, groups = {}; Object.keys(cachedData).forEach(k => { const item = cachedData[k]; if (item.date?.startsWith(month)) { grandTotal += item.amount; // --- 修正：相容舊資料的過濾邏輯 --- let isMatch = false; if (currentFilter === 'all') { isMatch = true; } else { // 如果有新版的 userIndex，直接比對 if (item.userIndex !== undefined) { isMatch = (item.userIndex === currentFilter); } // 如果是舊資料，比對文字名字 (與目前的按鈕名稱比對) else if (item.user !== undefined) { const targetName = (currentFilter === 1) ? uName1 : uName2; isMatch = (item.user === targetName); } } if (isMatch) { filteredTotal += item.amount; if (!groups[item.date]) groups[item.date] = []; groups[item.date].push({ ...item, id: k }); } } }); // 預算條與總額顯示 (保持不變) const percent = Math.min((grandTotal / monthlyBudget) * 100, 100).toFixed(1); document.getElementById('progress-bar').style.width = percent + '%'; document.getElementById('percent-label').innerText = `${percent}%`; document.getElementById('budget-summary').innerText = `剩餘：$${Math.max(monthlyBudget-grandTotal, 0).toLocaleString()}`; const filterName = currentFilter === 'all' ? '月' : (currentFilter === 1 ? uName1 : uName2); document.getElementById('totalDisplay').innerText = `${filterName}支出：$${filteredTotal.toLocaleString()}`; // 清單渲染 (顯示名稱也加入相容邏輯) Object.keys(groups).sort((a, b) => new Date(b) - new Date(a)).forEach(date => { let dayHtml = '', dayTotal = 0; groups[date].sort((a,b)=>b.ts - a.ts).forEach(item => { dayTotal += item.amount; // 決定顯示的名字 let displayName = ""; if (item.userIndex !== undefined) { displayName = (item.userIndex === 2) ? uName2 : uName1; } else { displayName = item.user || "未知"; // 顯示舊有的名字 } dayHtml += `
${item.emoji} ${item.category} $${item.amount} ${displayName}
${item.note ? `
💬 ${item.note}
` : ''}
`; }); list.innerHTML += `
📅 ${date.slice(5)}$${dayTotal}
${dayHtml}`; }); }
