function renderData() { if (!cachedData) return; const month = document.getElementById('monthPicker').value; const list = document.getElementById('historyList'); list.innerHTML = ''; let grandTotal = 0, filteredTotal = 0, groups = {}; let cats = { '食':0, '衣':0, '住':0, '行':0, '樂':0, '繳費':0, '超市':0, '網購':0 }; Object.keys(cachedData).forEach(k => { const item = cachedData[k]; if (item.date?.startsWith(month)) { grandTotal += item.amount; // 過濾邏輯保持不變... let isMatch = (currentFilter === 'all'); if (!isMatch) { if (item.userIndex !== undefined) isMatch = (item.userIndex === currentFilter); else if (item.user) isMatch = (item.user === (currentFilter === 1 ? uName1 : uName2)); } if (isMatch) { filteredTotal += item.amount; if (cats.hasOwnProperty(item.category)) cats[item.category] += item.amount; if (!groups[item.date]) groups[item.date] = []; groups[item.date].push({ ...item, id: k }); } } }); if (myChart) { myChart.data.datasets[0].data = Object.values(cats); myChart.update(); } // --- 重點：動態顏色進度條 --- const percentNum = (grandTotal / monthlyBudget) * 100; const percentDisp = Math.min(percentNum, 100).toFixed(1); const progressBar = document.getElementById('progress-bar'); progressBar.style.width = percentDisp + '%'; document.getElementById('percent-label').innerText = `${percentDisp}%`; // 根據百分比改變顏色 if (percentNum < 50) { progressBar.style.background = "var(--safe)"; // 安全：粉紫 } else if (percentNum < 80) { progressBar.style.background = "#FFB3A7"; // 注意：橘黃 } else { progressBar.style.background = "#FF4D6D"; // 警告：紅色 // 如果超過 80%，可以加一個額外的震動提醒 if (percentNum >= 100) vibrate('heavy'); } // 顯示總計與餘額 const filterName = currentFilter === 'all' ? '月' : (currentFilter === 1 ? uName1 : uName2); document.getElementById('totalDisplay').innerText = `${filterName}支出：$${filteredTotal.toLocaleString()}`; // 渲染列表邏輯保持不變... Object.keys(groups).sort((a,b)=>new Date(b)-new Date(a)).forEach(date => { let dayHtml = '', dayTotal = 0; groups[date].sort((a,b)=>b.ts-a.ts).forEach(item => { dayTotal += item.amount; let dName = (item.userIndex === 2) ? uName2 : uName1; if (item.userIndex === undefined && item.user) dName = item.user; dayHtml += `
${item.emoji} ${item.category} $${item.amount} ${dName}
${item.note?`
💬 ${item.note}
`:''}
`; }); list.innerHTML += `
📅 ${date.slice(5)}$${dayTotal}
${dayHtml}`; }); }
